   566тЖТ   566тЖТ   566тЖТfunction SpendingPatterns({ expenses, incomes }: { expenses: any[]; incomes: any[] }) {
   567тЖТ   567тЖТ   567тЖТ  const patterns = useMemo(() => {
   568тЖТ   568тЖТ   568тЖТ    const now = new Date()
   569тЖТ   569тЖТ   569тЖТ    const thisMonth = expenses.filter(e => {
   570тЖТ   570тЖТ   570тЖТ      const d = new Date(e.date)
   571тЖТ   571тЖТ   571тЖТ      return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear()
   572тЖТ   572тЖТ   572тЖТ    })
   573тЖТ   573тЖТ   573тЖТ    const lastMonth = expenses.filter(e => {
   574тЖТ   574тЖТ   574тЖТ      const d = new Date(e.date)
   575тЖТ   575тЖТ   575тЖТ      return d.getMonth() === (now.getMonth() - 1 + 12) % 12 && d.getFullYear() === (now.getMonth() === 0 ? now.getFullYear() - 1 : now.getFullYear())
   576тЖТ   576тЖТ   576тЖТ    })
   577тЖТ   577тЖТ   577тЖТ
   578тЖТ   578тЖТ   578тЖТ    const thisTotal = thisMonth.reduce((s, e) => s + e.amount, 0)
   579тЖТ   579тЖТ   579тЖТ    const lastTotal = lastMonth.reduce((s, e) => s + e.amount, 0)
   580тЖТ   580тЖТ   580тЖТ    const changePercent = lastTotal > 0 ? Math.round(((thisTotal - lastTotal) / lastTotal) * 100) : 0
   581тЖТ   581тЖТ   581тЖТ
   582тЖТ   582тЖТ   582тЖТ    // Top spending day of week
   583тЖТ   583тЖТ   583тЖТ    const dayTotals: Record<string, number> = {}
   584тЖТ   584тЖТ   584тЖТ    const dayNames = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat']
   585тЖТ   585тЖТ   585тЖТ    expenses.forEach(e => {
   586тЖТ   586тЖТ   586тЖТ      const day = dayNames[new Date(e.date).getDay()]
   587тЖТ   587тЖТ   587тЖТ      dayTotals[day] = (dayTotals[day] || 0) + e.amount
   588тЖТ   588тЖТ   588тЖТ    })
   589тЖТ   589тЖТ   589тЖТ    const topDay = Object.entries(dayTotals).sort((a, b) => b[1] - a[1])[0]
   590тЖТ   590тЖТ   590тЖТ
   591тЖТ   591тЖТ   591тЖТ    // Recurring vs one-time
   592тЖТ   592тЖТ   592тЖТ    const recurring = expenses.filter(e => e.isRecurring).reduce((s, e) => s + e.amount, 0)
   593тЖТ   593тЖТ   593тЖТ    const oneTime = expenses.filter(e => !e.isRecurring).reduce((s, e) => s + e.amount, 0)
   594тЖТ   594тЖТ   594тЖТ    const total = recurring + oneTime
   595тЖТ   595тЖТ   595тЖТ    const recurringPct = total > 0 ? Math.round((recurring / total) * 100) : 0
   596тЖТ   596тЖТ   596тЖТ
   597тЖТ   597тЖТ   597тЖТ    // Average transaction size
   598тЖТ   598тЖТ   598тЖТ    const avgTransaction = thisMonth.length > 0 ? thisMonth.reduce((s, e) => s + e.amount, 0) / thisMonth.length : 0
   599тЖТ   599тЖТ   599тЖТ
   600тЖТ   600тЖТ   600тЖТ    return { changePercent, topDay, recurringPct, avgTransaction, thisMonthCount: thisMonth.length }
   601тЖТ   601тЖТ   601тЖТ  }, [expenses, incomes])
   602тЖТ   602тЖТ   602тЖТ
   603тЖТ   603тЖТ   603тЖТ  const fmt = (n: number) => formatCurrency(n)
   604тЖТ   604тЖТ   604тЖТ
   605тЖТ   605тЖТ   605тЖТ  return (
   606тЖТ   606тЖТ   606тЖТ    <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-3">
   607тЖТ   607тЖТ   607тЖТ      <div className="glass-subtle rounded-xl p-3">
   608тЖТ   608тЖТ   608тЖТ        <p className="text-xs text-muted-foreground mb-1">Month-over-Month</p>
   609тЖТ   609тЖТ   609тЖТ        <div className="flex items-center gap-2">
   610тЖТ   610тЖТ   610тЖТ          {patterns.changePercent > 0 ? (
   611тЖТ   611тЖТ   611тЖТ            <TrendingUp className="w-4 h-4 text-rose-400" />
   612тЖТ   612тЖТ   612тЖТ          ) : patterns.changePercent < 0 ? (
   613тЖТ   613тЖТ   613тЖТ            <TrendingDown className="w-4 h-4 text-emerald-400" />
   614тЖТ   614тЖТ   614тЖТ          ) : (
   615тЖТ   615тЖТ   615тЖТ            <Minus className="w-4 h-4 text-amber-400" />
   616тЖТ   616тЖТ   616тЖТ          )}
   617тЖТ   617тЖТ   617тЖТ          <span className={`text-lg font-bold ${patterns.changePercent > 0 ? 'text-rose-400' : patterns.changePercent < 0 ? 'text-emerald-400' : 'text-amber-400'}`}>
   618тЖТ   618тЖТ   618тЖТ            {patterns.changePercent > 0 ? '+' : ''}{patterns.changePercent}%
   619тЖТ   619тЖТ   619тЖТ          </span>
   620тЖТ   620тЖТ   620тЖТ        </div>
   621тЖТ   621тЖТ   621тЖТ        <p className="text-[10px] text-muted-foreground mt-1">{patterns.changePercent > 0 ? 'Spending increased' : patterns.changePercent < 0 ? 'Great! Spending decreased' : 'Spending stable'}</p>
   622тЖТ   622тЖТ   622тЖТ      </div>
   623тЖТ   623тЖТ   623тЖТ
   624тЖТ   624тЖТ   624тЖТ      <div className="glass-subtle rounded-xl p-3">
   625тЖТ   625тЖТ   625тЖТ        <p className="text-xs text-muted-foreground mb-1">Biggest Spending Day</p>
   626тЖТ   626тЖТ   626тЖТ        <p className="text-lg font-bold">{patterns.topDay?.[0] || 'N/A'}</p>
   627тЖТ   627тЖТ   627тЖТ        <p className="text-[10px] text-muted-foreground mt-1">{fmt(patterns.topDay?.[1] || 0)} total</p>
   628тЖТ   628тЖТ   628тЖТ      </div>
   629тЖТ   629тЖТ   629тЖТ
   630тЖТ   630тЖТ   630тЖТ      <div className="glass-subtle rounded-xl p-3">
   631тЖТ   631тЖТ   631тЖТ        <p className="text-xs text-muted-foreground mb-1">Recurring Expenses</p>
   632тЖТ   632тЖТ   632тЖТ        <p className="text-lg font-bold">{patterns.recurringPct}%</p>
   633тЖТ   633тЖТ   633тЖТ        <div className="mt-1.5 h-1.5 bg-white/5 rounded-full overflow-hidden">
   634тЖТ   634тЖТ   634тЖТ          <motion.div initial={{ width: 0 }} animate={{ width: `${patterns.recurringPct}%` }} transition={{ duration: 1, delay: 0.5 }}
   635тЖТ   635тЖТ   635тЖТ            className="h-full rounded-full bg-gradient-to-r from-amber-400 to-rose-400" />

   638тЖТ   638тЖТ   638тЖТ      </div>
