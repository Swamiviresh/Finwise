   471тЖТ   471тЖТfunction SpendingPatterns({ expenses, incomes }: { expenses: any[]; incomes: any[] }) {
   472тЖТ   472тЖТ  const patterns = useMemo(() => {
   473тЖТ   473тЖТ    const now = new Date()
   474тЖТ   474тЖТ    const thisMonth = expenses.filter(e => {
   475тЖТ   475тЖТ      const d = new Date(e.date)
   476тЖТ   476тЖТ      return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear()
   477тЖТ   477тЖТ    })
   478тЖТ   478тЖТ    const lastMonth = expenses.filter(e => {
   479тЖТ   479тЖТ      const d = new Date(e.date)
   480тЖТ   480тЖТ      return d.getMonth() === (now.getMonth() - 1 + 12) % 12 && d.getFullYear() === (now.getMonth() === 0 ? now.getFullYear() - 1 : now.getFullYear())
   481тЖТ   481тЖТ    })
   482тЖТ   482тЖТ
   483тЖТ   483тЖТ    const thisTotal = thisMonth.reduce((s, e) => s + e.amount, 0)
   484тЖТ   484тЖТ    const lastTotal = lastMonth.reduce((s, e) => s + e.amount, 0)
   485тЖТ   485тЖТ    const changePercent = lastTotal > 0 ? Math.round(((thisTotal - lastTotal) / lastTotal) * 100) : 0
   486тЖТ   486тЖТ
   487тЖТ   487тЖТ    // Top spending day of week
   488тЖТ   488тЖТ    const dayTotals: Record<string, number> = {}
   489тЖТ   489тЖТ    const dayNames = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat']
   490тЖТ   490тЖТ    expenses.forEach(e => {
   491тЖТ   491тЖТ      const day = dayNames[new Date(e.date).getDay()]
   492тЖТ   492тЖТ      dayTotals[day] = (dayTotals[day] || 0) + e.amount
   493тЖТ   493тЖТ    })
   494тЖТ   494тЖТ    const topDay = Object.entries(dayTotals).sort((a, b) => b[1] - a[1])[0]
   495тЖТ   495тЖТ
   496тЖТ   496тЖТ    // Recurring vs one-time
   497тЖТ   497тЖТ    const recurring = expenses.filter(e => e.isRecurring).reduce((s, e) => s + e.amount, 0)

   499тЖТ   499тЖТ    const total = recurring + oneTime
   500тЖТ   500тЖТ    const recurringPct = total > 0 ? Math.round((recurring / total) * 100) : 0
   501тЖТ   501тЖТ
   502тЖТ   502тЖТ    // Average transaction size
   503тЖТ   503тЖТ    const avgTransaction = thisMonth.length > 0 ? thisMonth.reduce((s, e) => s + e.amount, 0) / thisMonth.length : 0
   504тЖТ   504тЖТ
   505тЖТ   505тЖТ    return { changePercent, topDay, recurringPct, avgTransaction, thisMonthCount: thisMonth.length }
   506тЖТ   506тЖТ  }, [expenses, incomes])
   507тЖТ   507тЖТ
   508тЖТ   508тЖТ  const fmt = (n: number) => formatCurrency(n)
   509тЖТ   509тЖТ
   510тЖТ   510тЖТ  return (

   512тЖТ   512тЖТ      <div className="glass-subtle rounded-xl p-3">
   513тЖТ   513тЖТ        <p className="text-xs text-muted-foreground mb-1">Month-over-Month</p>
   514тЖТ   514тЖТ        <div className="flex items-center gap-2">
   515тЖТ   515тЖТ          {patterns.changePercent > 0 ? (
   516тЖТ   516тЖТ            <TrendingUp className="w-4 h-4 text-rose-400" />
   517тЖТ   517тЖТ          ) : patterns.changePercent < 0 ? (
   518тЖТ   518тЖТ            <TrendingDown className="w-4 h-4 text-emerald-400" />
   519тЖТ   519тЖТ          ) : (
   520тЖТ   520тЖТ            <Minus className="w-4 h-4 text-amber-400" />
   521тЖТ   521тЖТ          )}
   522тЖТ   522тЖТ          <span className={`text-lg font-bold ${patterns.changePercent > 0 ? 'text-rose-400' : patterns.changePercent < 0 ? 'text-emerald-400' : 'text-amber-400'}`}>
   523тЖТ   523тЖТ            {patterns.changePercent > 0 ? '+' : ''}{patterns.changePercent}%

   525тЖТ   525тЖТ        </div>
   526тЖТ   526тЖТ        <p className="text-[10px] text-muted-foreground mt-1">{patterns.changePercent > 0 ? 'Spending increased' : patterns.changePercent < 0 ? 'Great! Spending decreased' : 'Spending stable'}</p>

   528тЖТ   528тЖТ
   529тЖТ   529тЖТ      <div className="glass-subtle rounded-xl p-3">
   530тЖТ   530тЖТ        <p className="text-xs text-muted-foreground mb-1">Biggest Spending Day</p>
   531тЖТ   531тЖТ        <p className="text-lg font-bold">{patterns.topDay?.[0] || 'N/A'}</p>
   532тЖТ   532тЖТ        <p className="text-[10px] text-muted-foreground mt-1">{fmt(patterns.topDay?.[1] || 0)} total</p>

   549тЖТ   549тЖТ      </div>
   550тЖТ   550тЖТ    </div>
