    64→function FadeInWhenVisible({ children, delay = 0, className = '' }: { children: React.ReactNode; delay?: number; className?: string }) {
    65→  const ref = useRef(null)
    66→  const isInView = useInView(ref, { once: true, margin: '-50px' })
    67→  return (
    68→    <motion.div ref={ref} initial={{ opacity: 0, y: 30 }} animate={isInView ? { opacity: 1, y: 0 } : {}} transition={{ duration: 0.6, delay, ease: 'easeOut' }} className={className}>
    69→      {children}
    70→    </motion.div>
    71→  )
    72→}

    74→function FloatingParticles() {
    75→  const [mounted, setMounted] = useState(false)
    76→  useEffect(() => { setMounted(true) }, [])
    77→  const particles = mounted ? Array.from({ length: 25 }, (_, i) => ({
    78→    id: i,
    79→    x: Math.random() * 100,
    80→    y: Math.random() * 100,
    81→    size: Math.random() * 3 + 1.5,
    82→    duration: Math.random() * 20 + 15,
    83→    delay: Math.random() * 10,
    84→    opacity: Math.random() * 0.25 + 0.05,
    85→    color: ['#34d399', '#22d3ee', '#a78bfa', '#fbbf24'][Math.floor(Math.random() * 4)],
    86→  })) : []
    87→  return (
    88→    <div className="absolute inset-0 overflow-hidden pointer-events-none">
    89→      {particles.map(p => (
    90→        <motion.div key={p.id} className="absolute rounded-full"
    91→          style={{ left: `${p.x}%`, top: `${p.y}%`, width: p.size, height: p.size, opacity: p.opacity, background: p.color }}
    92→          animate={{ y: [0, -40, 0], x: [0, 20, 0], opacity: [p.opacity, p.opacity * 2.5, p.opacity] }}
    93→          transition={{ duration: p.duration, repeat: Infinity, delay: p.delay, ease: 'easeInOut' }}
    94→        />
    95→      ))}
    96→    </div>
    97→  )
    98→}

   100→function AnimatedCounter({ target, prefix = '', suffix = '', duration = 2 }: { target: number; prefix?: string; suffix?: string; duration?: number }) {
   101→  const [count, setCount] = useState(0)
   102→  const ref = useRef(null)
   103→  const inView = useInView(ref, { once: true, margin: '-50px' })
   104→
   105→  useEffect(() => {
   106→    if (!inView) return
   107→    let start = 0
   108→    const end = target
   109→    const step = end / (duration * 60)
   110→    const timer = setInterval(() => {
   111→      start += step
   112→      if (start >= end) { setCount(end); clearInterval(timer) }
   113→      else setCount(Math.floor(start))
   114→    }, 1000 / 60)
   115→    return () => clearInterval(timer)
   116→  }, [inView, target, duration])
   117→
   118→  return <span ref={ref}>{prefix}{count.toLocaleString()}{suffix}</span>
   119→}
